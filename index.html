ConsoleMessage NodeJS Library


ConsoleMessage is a Javascript library to write colored logs on console, it'll helpful to identify type of log message by looking at color. You can find more about this in npm. 
  npm install -g consolemessage
//logger.js

var console = require('consolemessage');

console.message('success message', SUCCESS);
console.message('debug message', DEBUG);
console.message('info message', INFO);
console.message('warning message', WARNING);
console.message('error message', ERROR);
console.message('crtitical message', CRITICAL);
//output

success message

debug message

info message

warning message

error message

critical message
